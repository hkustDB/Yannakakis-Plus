with
S508 as (select src as v2, dst as v5, rating as Z, rating as ZZ from Graph as S), 
SAgg366 as (select v2, max(ZZ) as ZZ from S508 group by v2), 
Rjoin762 as (select src as v1, dst as v2, rating as Z, Graph.rating+SAgg366.ZZ as ZZ from Graph, SAgg366 where Graph.dst=SAgg366.v2 order by ZZ DESC limit 128), 
R464 as (select v2, max(Z) as Z from Rjoin762 group by v2), 
S_prune909 as (select v2, v5, S508.Z as Z, ZZ from S508 join R464 using(v2) order by R464.Z+S508.ZZ DESC limit 128), 
S_join725 as (select v5, Rjoin762.Z + S_prune909.Z as Z, v2, v1 from Rjoin762 join S_prune909 using(v2) order by Z DESC limit 128)
select sum(v1+v2+v5+Z) from S_join725;
